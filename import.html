const params = new URLSearchParams(window.location.search); const gistID = params.get("gist"); const name = params.get("name"); if (name) { document.getElementById("listName").textContent = decodeURIComponent(name); } if (gistID) { fetch("https://api.github.com/gists/" + gistID) .then(r => r.json()) .then(gist => { const content = gist.files["list.json"].content; const encoded = btoa(unescape(encodeURIComponent(content))); document.getElementById("importBtn").href = "shopcart://import?data=" + encodeURIComponent(encoded); document.getElementById("importBtn").style.display = "block"; const parsed = JSON.parse(content); if (parsed.name) document.getElementById("listName").textContent = parsed.name; }) .catch(() => { document.getElementById("errorMsg").style.display = "block"; }); } else { document.getElementById("errorMsg").style.display = "block"; }
